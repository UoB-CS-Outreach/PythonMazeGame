# Python-subset sample: right-hand rule solver
# Allowed: if/elif/else, while, pass, and/or/not, the functions below
# Functions you can call:
#   move(), turn_left(), turn_right()
#   path_ahead(), path_left(), path_right(), at_goal()

while not at_goal():
    if path_right():
        turn_right()
        move()
    elif path_ahead():
        move()
    else:
        turn_left()
